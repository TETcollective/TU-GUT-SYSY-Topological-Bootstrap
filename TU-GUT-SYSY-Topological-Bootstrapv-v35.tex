import numpy as np
from scipy.integrate import odeint
import matplotlib.pyplot as plt

# Parameters (ultraclean regime)
N_vortices = 80
box_size = 10.0
nu = 1e-8                     # Extremely low viscosity
steps = 6000
np.random.seed(42)

positions = box_size * np.random.rand(N_vortices, 2)
circulations = np.random.choice([-1.0, 1.0], N_vortices)

def biot_savart_velocity(pos, all_pos, all_gamma):
    vel = np.zeros(2)
    for j in range(len(all_pos)):
        if np.all(all_pos[j] == pos):
            continue
        r = all_pos[j] - pos
        r -= box_size * np.round(r / box_size)
        dist2 = np.dot(r, r) + 1e-10
        vel += all_gamma[j] * np.array([-r[1], r[0]]) / (2 * np.pi * dist2)
    return vel

def equations(state, t):
    pos = state.reshape((N_vortices, 2))
    vel = np.zeros_like(pos)
    for i in range(N_vortices):
        vel[i] = biot_savart_velocity(pos[i], pos, circulations)
    return (vel - nu * pos).flatten()

def linking_number(curve1, curve2):
    curve1 = np.append(curve1, [curve1[0]], axis=0)
    curve2 = np.append(curve2, [curve2[0]], axis=0)
    d1 = np.diff(curve1, axis=0)
    d2 = np.diff(curve2, axis=0)
    lk = 0.0
    for i in range(len(d1)):
        for j in range(len(d2)):
            r = curve1[i] - curve2[j]
            r_norm = np.linalg.norm(r)
            if r_norm < 1e-8:
                continue
            cross_term = d1[i,0]*d2[j,1] - d1[i,1]*d2[j,0]
            lk += cross_term / (r_norm**3 + 1e-10)
    return np.abs(lk / (4 * np.pi))

state0 = positions.flatten()
t = np.linspace(0, 100, steps + 1)
trajectory = odeint(equations, state0, t, rtol=1e-8, atol=1e-10)

lk_history = []
sample_times = []
for k in range(0, steps + 1, 600):
    pos_k = trajectory[k].reshape((N_vortices, 2))
    curve1 = pos_k[0:20]
    curve2 = pos_k[20:40]
    lk = linking_number(curve1, curve2)
    lk_history.append(lk)
    sample_times.append(t[k])

print(f"Linking number std dev: {np.std(lk_history):.2e}")

plt.figure(figsize=(14,9), facecolor='black')

plt.ylabel("Spatial Coordinate Y", color='white')

plt.tight_layout()

plt.close()
print("Visualization saved.")
